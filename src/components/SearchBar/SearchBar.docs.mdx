# SearchBar

A professional search bar component with debounce capabilities.

## Usage

```tsx
import { SearchBar } from './SearchBar';

const MyComponent = () => {
  const handleSearch = (query: string) => {
    console.log('Searching for:', query);
  };

  return <SearchBar onSearch={handleSearch} />;
};
```
